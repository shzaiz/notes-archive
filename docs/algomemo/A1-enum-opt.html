4.滑动窗口

核心思想. 与其总是从全新生成新的知识, 不如复用以往的知识加以改动.

知识简介. 通常是维护一段连续区间的知识, 通过仅仅考虑新元素的加入和旧元素的清除线性地维护现有的知识.

例 4.1. LC2090.半径为 k 的子数组平均值.

  1. 连续区间维护的知识是什么? 这个窗口有哪些约束条件?

  2. 在头部的元素删除、头部元素添加的时候, 应该做出怎样的维护?

  3. 实现对应的代码. 一个循环就是一个归纳过程.

代码演示

C++: 

class Solution {
public:
    vector<int> getAverages(vector<int>& nums, int k) {
        vector<int > ans;
        long long klen = 2 * k + 1LL;
        int sz = nums.size();
        if(klen > sz){
            for(int i=0; i<sz; i++) ans.push_back(-1);
            return ans;
        }

        for(int i=0; i<k; i++) ans.push_back(-1);
        
        long long curr = [&]() -> long long{
            long long ans = 0;
            for(int i=0; i<klen; i++){
                ans += nums[i];
            }
            return ans;
        }();
        ans.push_back(curr/klen);
        // 到这里: 已经得到了 nums[0, klen)的结果
        // 这个循环: nums[lp-1, rp-1) 的结果 -> nums[lp, rp)的结果
        for(int lp=1; lp<sz-k; lp++){
            int rp = lp + klen; // [lp, rp)
            if(rp > sz) {ans.push_back(-1); continue;}
            // 加入rp
            curr += nums[rp-1];
            // 抽走lp
            curr -= nums[lp-1];
            // 维护答案
            ans.push_back(curr / klen);
        }
        
        return ans;
    }
};

例 4.2. LC30. 串联所有单词的子串.

  1. 既然不关心顺序, 那么就可以仅仅记录一段序列中是否“出现”.

  2. 如果类似上例一样一个一个格子地移动, 就不好处理了.

  3. 鉴于每个字符都是等长的特性, 可以每次移动那么多, 并分那么多个窗口一起滑动.

代码实现

C++: 

class Solution {
public:
    vector<int> findSubstring(string s, vector<string>& words) {
        int n_elm = words.size();
        int sz_str = words[0].size();
        vector<int> ans;
        map<string, int> target; 
        for(auto s : words){
            target[s]++;
        }

        
        vector<string> split; 
        for(int offset=0; offset < sz_str; offset++){
            split.clear();

            for(int i=offset; i+sz_str<=s.size(); i+=sz_str){
                split.push_back(s.substr(i, sz_str));
            }

            int lptr = 0, rptr = n_elm-1;
            
            map<string, int> fact;
            for(int i=lptr; i<=rptr; i++){
                if(i >= split.size()) break; // 注意边界条件
                fact[split[i]] += 1;
            }
            while(rptr < split.size()){
                if(fact == target) ans.push_back(lptr*sz_str + offset);
                if(rptr == split.size()-1) break; // 注意边界条件
                
                rptr++; fact[split[rptr]]+=1;
                
                fact[split[lptr]]--;
                if(fact[split[lptr]] == 0) fact.erase(split[lptr]);
                lptr++;
            }
        }
        return ans;
    }
};

例 4.3. LC2799. 统计完全子字符串的数目.

  1. 枚举右端点, 得到左侧第一个不满足的位置.

  2. 注意单调性: 子数组不重复的元素不可能多于整个元素的, 并且一个短的如果有一定的元素, 长的一定有.

代码实现:

C++ 

class Solution {
public:
    int countCompleteSubarrays(vector<int>& nums) {
        int n = nums.size();
        unordered_set<int> all(nums.begin(), nums.end());
        int k = all.size();  // 总共不同元素个数

        unordered_map<int, int> cnt;
        int ans = 0, left = 0;
        for (int right = 0; right < n; ++right) {
            cnt[nums[right]]++;

            while (cnt.size() == k) {
                cnt[nums[left]]--;
                if (cnt[nums[left]] == 0) {
                    cnt.erase(nums[left]);
                }
                left++;
            }
            ans += left;
        }

        return ans;
    }
};

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> ans;
        ranges::sort(nums);
        int lptr = 0, rptr = nums.size()-1;
        for(int _=0; _<nums.size(); _++){
            int elem = nums[_];
            if(_ > 0 && elem == nums[_-1]) {continue;}
            lptr = _+1, rptr = nums.size()-1;
            while(lptr < rptr){
                int cval = nums[lptr] + nums[rptr] +elem;
                if(cval == 0){
                    vector<int> anss = {elem, nums[lptr], nums[rptr]};
                    ans.push_back(anss);
                    for(lptr++; lptr<rptr && nums[lptr] == nums[lptr-1]; lptr++);
                }else if(cval > 0){
                    rptr--;
                }else{
                    lptr++;
                }
                
            }
        }
        return ans;
    }
};

例 5.2. 2563. 统计公平数对的数目

答案. 排序之后的结果: 两个指针越靠后, 答案越大; 满足单调性相关: 区间越长, 答案越大.

  • 枚举一个数, 决定左右两边的范围

代码演示 

class Solution {
public:
// 0 1 4 4 5 7
// ~   L   R   --> 3
//   ~ L   R   --> 2

// 1 2 5 7 9 [11, 11]
// ~       LR
//   ~     LR
    long long countFairPairs(vector<int>& nums, int lower, int upper) {
        // from smallest to biggest
        sort(nums.begin(), nums.end());
        long long ans = 0;
        int n = nums.size();
        int rLoPtr = n-1, rHiPtr = n-1;
        for(int lptr = 0; lptr < n; lptr++){
            if((lptr > rLoPtr)) {
                rLoPtr = max(rLoPtr, lptr);
                rHiPtr = max(rHiPtr, lptr);
            }
            
            while((rLoPtr > lptr) && 
                                  (nums[lptr] + nums[rLoPtr] >= lower)) 
                rLoPtr--;
            while((rHiPtr > lptr) && 
                                  (nums[lptr] + nums[rHiPtr] > upper)) 
                rHiPtr--;
               
            ans += (rHiPtr - rLoPtr);
        }
        return ans;
    }
};

例 5.3. 611. 有效三角形的个数

答案. 可以枚举最短边; 由于递增的特性, 只要最大的两个边小于第三边即可.

代码演示

Python3: 

class Solution:
    def triangleNumber(self, nums: List[int]) -> int:
        nums.sort()
        n, ans = len(nums), 0
        for iptr in range(n):
            ival = nums[iptr]
            if ival == 0: continue
            # 枚举jptr, kptr
            jptr, kptr = iptr+1, iptr+2
            if jptr >= n or kptr >= n: continue
            for kptr in range(iptr+2, n):
                while nums[kptr] - nums[jptr] >= nums[iptr]:
                    jptr+=1
                ans += kptr-jptr
        return ans

例 6.1. 给一个按照非递减顺序排列的整数数组和一个目标值. 请你找出给定目标值在数组中的开始位置和结束位置. (34. 在排序数组中查找元素的第一个和最后一个位置)

解答. 考虑仅仅实现函数lower_bound, 是最小满足的下标. 这可以做到开头的位置. 那么最后一个可以考虑求lower_bound(t+1), 然后位置就好了. 当然也可以把两个都实现了, 就像下面的代码一样.

函数

l 左侧元素

r 右侧元素

ans 条件

findFirst

全都 < target

全都 ≥ target

nums[ans] == target

findLast

全都 ≤ target

全都 > target

nums[ans] == target

代码实现

C++ 

class Solution {
public:

    int findFirst(const vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1, ans = -1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            //nums[mid] >= target: 目标值可能还在左边（包括当前的 mid），所以我们移动右边界
            if (nums[mid] >= target) right = mid - 1;
            //nums[mid] < target: 目标值只能在右边，移动左边界：l = mid + 1
            else left = mid + 1;
            //nums[mid] == target: 更新结果 ans = mid. 但不立即返回
            if (nums[mid] == target) ans = mid;
        }
        return ans;
    }

    int findLast(const vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1, ans = -1;
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] <= target) left = mid + 1;
            else right = mid - 1;
            if (nums[mid] == target) ans = mid;
        }
        return ans;
    }

    vector<int> searchRange(vector<int>& nums, int target) {
        return { findFirst(nums, target), findLast(nums, target) };
    }
};
