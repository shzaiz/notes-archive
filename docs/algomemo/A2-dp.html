class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        n, m = len(nums), sum(nums)
        f = [[False] * m for _ in range(n)]
        f[0][0] = True
        for i in range(1, n):
            for j in range(0, m):
                f[i][j] = f[i-1][j] | f[i-1][j-nums[i]]

        if m%2 == 1: return False
        else: return f[n-1][m//2]
