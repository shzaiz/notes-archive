class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        // f[i][j] ::= t1的前i个字符和t2的前j个字符的LCS
        // f[i][j] = max(f[i-1][j-1]+(a[i]==b[j]), f[i-1][j], f[i][j-1])

        vector<vector<int>> f(text1.size(), vector<int>(text2.size(), 0));
        for(int i=0; i<text1.size(); i++){
            for(int j=0; j<text2.size(); j++){
                int sameq = (text1[i]==text2[j]?1:0);
                if(sameq) {
                    if(i==0 || j==0) f[i][j]=1;
                    if(i-1 >= 0 && j-1>=0) f[i][j] = f[i-1][j-1]+1;
                }
                if(i-1 >= 0) f[i][j] = max(f[i][j], f[i-1][j]);
                if(j-1 >= 0) f[i][j] = max(f[i][j], f[i][j-1]);
            }
        }
        return f[text1.size()-1][text2.size()-1];

    }
};

class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        vector<int> f(n);
        ranges::fill(f, 1);
        f[0] = 1;
        for(int pos=1; pos<n; pos++){
            for(int i=0; i<pos; i++){
                if(nums[i]<nums[pos]){
                    f[pos] = max(f[pos], f[i] + 1);
                }
            }
        }
        return *max_element(f.begin(), f.end());
    }
};

例 1.5. (分割回文串) 给定一个字符串, 请分为一些子串, 使每个子串都是回文串, 返回符合要求的最小分割次数. (132. 分割回文串 II)

解答. 可以考虑把切分为回文串, 最小切分的次数 

class Solution:
    def minCut(self, s: str) -> int:
        def ispar(s:str) -> bool:
            return s[::-1]==s
        def f(r:int) -> int:
            # [l..r]是回文串
            if ispar(s[0:r+1]):
                return 0
            # [l..r]不是回文串, 枚举切分点, 每次增加一个
            res = inf 
            for l in range(1, r+1):
                if(ispar(s[l:r+1])):
                    res = min(res, f(l-1)+1)
            return res
        return f(len(s)-1)

练习 1.2. 2707. 字符串中的额外字符

解答. 定义为以需要丢弃多少个字符. 

class Solution:
    def minExtraChar(self, s: str, dictionary: List[str]) -> int:
        @cache
        def dfs(r:int) -> int: 
            if r<0: return 0
            res = dfs(r-1)+1 # 不选, 丢掉的字母+1
            for j in range(r+1):
                # 选择的条件: 在字典里面
                if s[j:r+1] in dictionary:
                    res = min(res, dfs(j-1))
            return res
        return dfs(len(s)-1)

class Solution:
    def maximumTotalCost(self, nums: List[int]) -> int:
        
        num = list(accumulate(map(lambda x:-x[1] if x[0]%2 else x[1], enumerate(nums))))
        
        def get(l:int, r:int) -> int:
            coeff = 1
            if l % 2 : coeff *= -1
            # if (r-l+1) % 2: coeff *= -1
            if(l==0): return num[r] * coeff
            else: return (num[r]-num[l-1]) * coeff

        @cache
        def f(r:int) -> int:
            if r<0: return 0
            if r==0: return num[0] 
            # for l=r
            res = get(r, r) + f(r-1)
            
            # for l=1..r-1
            for l in range(0, r):
                currans = get(l, r) + f(l-1)
                res = max(res, currans)
            return res
        
        return f(len(nums)-1)
